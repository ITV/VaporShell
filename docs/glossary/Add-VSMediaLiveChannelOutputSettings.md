 [[-MsSmoothOutputSettings] <Object>] [[-FrameCaptureOutputSettings] <Object>] [[-HlsOutputSettings] <Object>]
 [[-RtmpOutputSettings] <Object>] [[-UdpOutputSettings] <Object>] [[-MultiplexOutputSettings] <Object>]
